# 백준 #1992 [쿼드트리](https://www.acmicpc.net/problem/1992)
`실버1` `재귀` `분할 정복`
---
- 재귀로 짜야겠다고 떠오르긴 했는데, 실제 구현이 어려웠다.
- 시작행,시작열,행,열을 인자로 넘겼는데, 이걸 2로 계속 나누다보면 1단위로 출력해야할 때 걸렸다.

### 처음코드
```jsx
vector<string>v; //트리 정보
vector<char>ans; //정답

void solve(int sr, int sc, int r, int c) { //시작행, 시작열, 전체행, 전체열
	ans.push_back('(');
	char init;//시작문자
	bool flag = true; //걸린거 있는지 여부
	if (r-sr != 2) { //한칸 단위 검사 아니면
    //좌상단 탐색
		for (int i = sr; i < r / 2; i++) {
			for (int j = sc; j < c / 2; j++) {
				if (i == 0 && j == 0)
					init = v[i][j];
				else {
					if (v[i][j] != init) //시작문자와 같지 않으면
					{
						solve(sr, sc, r / 2, c / 2); //재귀호출
						flag = false;
						break;
					}
				}
			}
			if (!flag)
				break;
		}
		if (flag) //걸린거 없으면 정답에 초기문자 담음
			ans.push_back(init);
		flag = true;
    //우상단 탐색
		for (int i = sr; i < r / 2; i++) {
			for (int j = c / 2; j < c; j++) {
				if (i == 0 && j == c / 2)
					init = v[i][j];
				else {
					if (v[i][j] != init)
					{
						solve(r / 2, sc, r, c / 2);
						flag = false;
						break;
					}
				}
			}
			if (!flag)
				break;
		}
		if (flag)
			ans.push_back(init);
		flag = true;
    //좌하단 탐색
		for (int i = r / 2; i < r; i++) {
			for (int j = sc; j < c / 2; j++) {
				if (i == r / 2 && j == 0)
					init = v[i][j];
				else {
					if (v[i][j] != init)
					{
						solve(sr, c/2, r, c / 2);
						flag = false;
						break;
					}
				}
			}
			if (!flag)
				break;
		}
		if (flag)
			ans.push_back(init);
		flag = true;
    //우하단 탐색
		for (int i = r / 2; i < r; i++) {
			for (int j = c / 2; j < c; j++) {
				if (i == r / 2 && j == c / 2)
					init = v[i][j];
				else {
					if (v[i][j] != init)
					{
						solve(r / 2, c / 2, r, c);
						flag = false;
						break;
					}
				}
			}
			if (!flag)
				break;
		}
		if (flag)
			ans.push_back(init);
	}
	else { //1칸단위 탐색일 때는 모두 담음
		init = v[sr][sc];
		ans.push_back(init);
		ans.push_back(v[sr][sc + 1]);
		ans.push_back(v[sr + 1][sc]);
		ans.push_back(v[sr + 1][sc + 1]);
	}
	
	ans.push_back(')');
}

int main()
{
	int N;
	cin >> N;
	for (int i = 0; i < N; i++) {
		string s;
		cin >> s;
		v.push_back(s);
	}
	solve(0, 0, N, N); //
	for (int i = 0; i < ans.size(); i++)
		cout << ans[i];
}
```
- 구현 못하겠어서 타 코드 참고함.
- 코드 보니 너무 어렵게 접근한걸 알았음. for 문의 범위도 잘못 설정했음.
### 전체코드
```jsx
string v[64];

void solve(int size, int r, int c) { //사이즈, 행, 열
	char current = v[r][c]; //현재 위치의 문자

  //행, 열과 사이즈에 따라 for 탐색
	for (int i = r; i < r + size; i++) {
		for (int j = c; j < c + size; j++) {
			if (v[i][j] != current) { //같지않을 때
				cout << "(";
        //4방위 모두 재귀호출
				solve(size / 2, r, c);
				solve(size / 2, r, c + size / 2);
				solve(size / 2, r + size / 2, c);
				solve(size / 2, r + size / 2, c + size / 2);
				cout << ")";
				return;
			}
		}
	}
	cout << current;
}

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);

	int N;
	cin >> N;
	for (int i = 0; i < N; i++) 
		cin >> v[i];
	
	solve(N,0,0);
}
```
